with_no_such_dependency_error_handling() { || ... } Show source 
# File lib/bundler/vendor/molinillo/lib/molinillo/delegates/specification_provider.rb, line 69
def with_no_such_dependency_error_handling
  yield
rescue NoSuchDependencyError => error
  if state
    vertex = activated.vertex_named(name_for(error.dependency))
    error.required_by += vertex.incoming_edges.map { |e| e.origin.name }
    error.required_by << name_for_explicit_dependency_source unless vertex.explicit_requirements.empty?
  end
  raise
end

Ensures any raised {NoSuchDependencyError} has its {NoSuchDependencyError#required_by} set. @yield
