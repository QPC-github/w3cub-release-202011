module ERB::Util

def url_encode(s)
  s.to_s.b.gsub(/[^a-zA-Z0-9_\-.~]/n) { |m|
    sprintf("%%%02X", m.unpack1("C"))
  }
end

A utility method for encoding the String s as a URL.
