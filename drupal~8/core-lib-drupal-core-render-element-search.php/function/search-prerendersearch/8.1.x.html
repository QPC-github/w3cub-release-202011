Parameters

public static function preRenderSearch($element) {
  $element['#attributes']['type'] = 'search';
  Element::setAttributes($element, array('id', 'name', 'value', 'size', 'maxlength', 'placeholder'));
  static::setAttributes($element, array('form-search'));

  return $element;
}
