public function release($name) {
  unset($this->locks[$name]);
  try {
    $this->database->delete('semaphore')
      ->condition('name', $name)
      ->condition('value', $this->getLockId())
      ->execute();
  }
  catch (\Exception $e) {
    $this->catchException($e);
  }
}
